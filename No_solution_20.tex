\section*{Solution Summaries}

\textbf{Q1.} Exponential growth every 6 hours by 1.29.  
\(A = 34.11(1.29)^{x/6}\). → \textbf{A}

\textbf{Q2.} \(\frac{\sin X}{\cos Y} = 1 \Rightarrow X = 90° - Y = 58°.\) → \textbf{B}

\textbf{Q3.} Parallel line geometry; result \(w = 410.\)

\textbf{Q4.} Convert yearly growth (1.02) to quarterly: \((1.02)^{q/4}\). → \textbf{A}

\textbf{Q5.} Product of roots = \(-24st = -2kst \Rightarrow k=12.\)

\textbf{Q6.} \(a=1.4 \Rightarrow y=900(1.4)(3.5)^2=15{,}435.\)

\textbf{Q7.} Use cylinder shell volume: \(V = \pi h(R^2-r^2) \approx 19{,}340.\) → \textbf{D}

\textbf{Q8.} Similar trapezoids → corresponding angles equal. → \textbf{B}

\textbf{Q9.} \(\pi/15 = 12° \Rightarrow n=2.\)

\textbf{Q10.} Slope \(9/0.18 = 50 \Rightarrow t=50d.\) → \textbf{A}

\textbf{Q11.} Linear depreciation: \(p = 750 - 135t.\) → \textbf{B}

\textbf{Q12.} \(9n \ge 1170.\) → \textbf{D}

\textbf{Q13.} 62 sedans = 20\% → total 310 → 30\% minivans = 93.

\textbf{Q14.} \(1.02 \pm 0.05 \Rightarrow 0.97 \le w \le 1.07.\) → \textbf{C}

\textbf{Q15.} \(3(5p)+2(4s)=136 \Rightarrow 15p+8s=136.\) → \textbf{D}

\textbf{Q16.} Only local sample, not nationwide → \textbf{D.}

\textbf{Q17.} Weighted mean \(=88.\) → \textbf{B.}

\textbf{Q18.} Removing outlier (19) doesn’t change median → \textbf{B.}

\textbf{Q19.} Scale factor = 3 → area factor = 9 → \(n-k=1680.\) → \textbf{C.}
